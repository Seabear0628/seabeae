local function OYFOAZD_fake_script() -- LoginFrame.LoginCloseScript 
	local script = Instance.new('Script', LoginFrame)

	local LoginFrame = script.Parent
	local LoginClose = LoginFrame:WaitForChild("LoginClose")
	local LoginOpen = LoginFrame.Parent:WaitForChild("LoginOpen")
	
	LoginOpen.MouseButton1Click:Connect(function()
	 LoginOpen.Visible = false
	 LoginFrame.Visible = true
	end)
	
	LoginClose.MouseButton1Click:Connect(function()
	 LoginOpen.Visible = true
	 LoginFrame.Visible = false
	end)
end
